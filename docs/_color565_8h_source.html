283 inline color565 to565(float_t H, float_t S, float_t V) {
284 float_t p, q, t, ff;
285 int8_t i;
286 float_t r, g, b;
287
288 if (H >= 360.0) H = 0.0;
289 H /= 60.0;
290 i = (int8_t)H;
291 ff = H - i;
292 p = V * (1.0 - S);
293 q = V * (1.0 - (S * ff));
294 t = V * (1.0 - (S * (1.0 - ff)));
295
296 switch (i) {
297 case 0: r = V; g = t; b = p; break;
298 case 1: r = q; g = V; b = p; break;
299 case 2: r = p; g = V; b = t; break;
300 case 3: r = p; g = q; b = V; break;
301 case 4: r = t; g = p; b = V; break;
302 case 5:
303 default:
304 r = V; g = p; b = q; break;
305 }
306
307 return ((uint8_t)(r * 31.0f) << 11) | ((uint8_t)(g * 63.0f) << 5) | (uint8_t)(b * 31.0f);
308 }
