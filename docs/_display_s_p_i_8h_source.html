66 // Write pixels. Some SPI implementations require the final pixel
67 // to be written differently, so we need to keep track of the count
68 if (_px == 1) {
69 int lineOffset = 0;
70 int count = buffer.rect.width * buffer.rect.height;
71 for (uint16_t y = buffer.rect.y; y <= buffer.rect.y2; y++) {
72 for (uint16_t x = buffer.rect.x; x <= buffer.rect.x2; x++) {
73 if (--count) writeData16(to565(buffer.pixels[lineOffset + x]));

271 ALWAYS_INLINE void commandSequence(const uint8_t* commands) {
272 const uint8_t* addr = commands;
273 uint8_t c;
274 while (1) {
275 uint8_t count = *addr++;
276 if (count-- == 0) break;
277 c = *addr++;
278 if (c == DELAY_COMMAND) {
279 c = *addr++;
280 delay(c);

282 else {
283 writeCommand(c);
284 while (count-- > 0) {

387 ALWAYS_INLINE void waitFifoNotFull(void) {
388 uint32_t sr;
389 uint32_t tmp __attribute__((unused));
390 do {
391 sr = KINETISK_SPI0.SR;
392 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR; // drain RX FIFO
393 } while ((sr & (15 << 12)) > (3 << 12));
394 }

399 ALWAYS_INLINE void waitFifoEmpty(void) {
400 uint32_t sr;
401 uint32_t tmp __attribute__((unused));
402 do {
403 sr = KINETISK_SPI0.SR;
404 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR; // drain RX FIFO
405 } while ((sr & 0xF0F0) > 0); // wait both RX & TX empty
406 }

408 ALWAYS_INLINE void waitTransmitComplete(void) {
409 uint32_t tmp __attribute__((unused));
410 while (!(KINETISK_SPI0.SR & SPI_SR_TCF)); // wait until final output done
411 tmp = KINETISK_SPI0.POPR; // drain the final RX FIFO word
412 }
413
414 ALWAYS_INLINE void waitTransmitComplete(uint32_t mcr) {
415 uint32_t tmp __attribute__((unused));
416 while (1) {
417 uint32_t sr = KINETISK_SPI0.SR;
418 if (sr & SPI_SR_EOQF) break; // wait for last transmit
419 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR;
420 }
421 KINETISK_SPI0.SR = SPI_SR_EOQF;
422 SPI0_MCR = mcr;
423 while (KINETISK_SPI0.SR & 0xF0) {
424 tmp = KINETISK_SPI0.POPR;
425 }
426 }
