236 uint32_t sr;
237 uint32_t tmp __attribute__((unused));
238 do {
239 sr = KINETISK_SPI0.SR;
240 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR; // drain RX FIFO
241 } while ((sr & (15 << 12)) > (3 << 12));

247 ALWAYS_INLINE void waitFifoEmpty(void) {
248 uint32_t sr;
249 uint32_t tmp __attribute__((unused));
250 do {
251 sr = KINETISK_SPI0.SR;
252 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR; // drain RX FIFO
253 } while ((sr & 0xF0F0) > 0); // wait both RX & TX empty
254 }

256 ALWAYS_INLINE void waitTransmitComplete(void) {
257 uint32_t tmp __attribute__((unused));
258 while (!(KINETISK_SPI0.SR & SPI_SR_TCF)); // wait until final output done
259 tmp = KINETISK_SPI0.POPR; // drain the final RX FIFO word
260 }

262 ALWAYS_INLINE void waitTransmitComplete(uint32_t mcr) {
263 uint32_t tmp __attribute__((unused));
264 while (1) {
265 uint32_t sr = KINETISK_SPI0.SR;
266 if (sr & SPI_SR_EOQF) break; // wait for last transmit
267 if (sr & 0xF0) tmp = KINETISK_SPI0.POPR;
268 }
269 KINETISK_SPI0.SR = SPI_SR_EOQF;
270 SPI0_MCR = mcr;
271 while (KINETISK_SPI0.SR & 0xF0) {
272 tmp = KINETISK_SPI0.POPR;
273 }
274 }
