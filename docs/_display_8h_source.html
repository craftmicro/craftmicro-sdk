66 setDestinationArea(
67 buffer.rect.x << _px, // top-left x
68 buffer.rect.y << _py, // top-left y

70 ((buffer.rect.y2 + 1) << _px) - 1 // bottom-right y
71 );
72 */
73
74 // (4) Now send the data
75 // Pixel scaling allows 1 pixel in the data to be scaled up to 2x2, 4x4 or 8x8 pixels
76 // on the display. That means that if a line of pixels in scaled by 4x4 we actually
77 // need to draw that line 4 times. Likewise, we also need to draw each pixel
78 // horizontally 4 times.
79 // Example: data[ A, B, C ] at 4x4 pixel scale (_px)
80 // Display: AAAABBBBCCCC
81 // AAAABBBBCCCC
82 // AAAABBBBCCCC
83 // AAAABBBBCCCC
84 //
85 // Colours are always provided in 888 format (0x00000000RRRRRRRRGGGGGGGGBBBBBBBB). Use
86 // the conversion functions for the hardware's colour. For example to565
87 /*
88 int scale = 1 << _px;
89 int lineOffset = buffer.rect.y * _size.width;
90 for ( uint16_t y = buffer.rect.y; y <= buffer.rect.y2; y++ ) {
91 for ( uint16_t yi = 0; yi < scale; yi++ ) {
92 for ( uint16_t x = buffer.rect.x; x <= buffer.rect.x2; x++ ) {
93 for ( uint16_t xi = 0; xi < scale; xi++ ) {
94 sendPixel( to565( buffer.pixels[lineOffset + x] ) );
95 }
96 }
97 }

99 }
100 */
101
102 // (5) Finalise the process to stop sending data to the hardware
103 /*
104 endTransaction();
105 */
106
107 // (6) Set ready flag to indicate we are ready for the next draw call
