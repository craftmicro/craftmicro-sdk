64 // (1) Set ready flag to indicate we are not ready for the next draw call yet.
65 // Note that this is not really required for syncronous methods (like this example)
66 // but is important if DMA or an alternative asyncronous method of sending the
67 // pixel data is used.
68 ready = false;
69
70 // (2) Initilise process for sending data to the hardware
71 /*
72 startTransaction();
73 */
74
75 // (3) Hardware will first usually require the user to specify the screen area to draw to
76 /*
77 setDestinationArea(
78 buffer.rect.x << _px, // top-left x
79 buffer.rect.y << _py, // top-left y
80 ((buffer.rect.x2 + 1) << _px) - 1, // bottom-right x
81 ((buffer.rect.y2 + 1) << _px) - 1 // bottom-right y
82 );
83 */
84
85 // (4) Now send the data
86 // Pixel scaling allows 1 pixel in the data to be scaled up to 2x2, 4x4 or 8x8 pixels
87 // on the display. That means that if a line of pixels in scaled by 4x4 we actually
88 // need to draw that line 4 times. Likewise, we also need to draw each pixel
89 // horizontally 4 times.
90 // Example: data[ A, B, C ] at 4x4 pixel scale (_px)
91 // Display: AAAABBBBCCCC
92 // AAAABBBBCCCC
93 // AAAABBBBCCCC
94 // AAAABBBBCCCC
95 //
96 // Colours are always provided in 888 format (0x00000000RRRRRRRRGGGGGGGGBBBBBBBB). Use
97 // the conversion functions for the hardware's colour. For example to565
98 /*
99 int scale = 1 << _px;
100 int lineOffset = buffer.rect.y * _size.width;
101 for ( uint16_t y = buffer.rect.y; y <= buffer.rect.y2; y++ ) {
102 for ( uint16_t yi = 0; yi < scale; yi++ ) {
103 for ( uint16_t x = buffer.rect.x; x <= buffer.rect.x2; x++ ) {
104 for ( uint16_t xi = 0; xi < scale; xi++ ) {
105 sendPixel( to565( buffer.pixels[lineOffset + x] ) );
106 }
107 }
108 }
109 lineOffset += _size.width;
110 }
111 */
112
113 // (5) Finalise the process to stop sending data to the hardware
114 /*
115 endTransaction();
116 */
117
118 // (6) Set ready flag to indicate we are ready for the next draw call
119 ready = true;
120 }
