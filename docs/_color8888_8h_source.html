268 inline color8888 to8888(float_t h, float_t s, float_t v) {
269 int i;
270 float f, p, q, t;
271 float r, g, b;
272 v = alphaClamp(v);
273 s = alphaClamp(s);
274 h = fmod(h, 360.0f);
275 if (h < 0) h = 360.0f + h;
276 if (s == 0) {
277 // achromatic (grey)
278 r = g = b = v;
279 return ((uint8_t)(r * 255.0f) << 16) | ((uint8_t)(g * 255.0f) << 8) | (uint8_t)(b * 255.0f);
280 }
281 h /= 60; // sector 0 to 5
282 i = floor(h);
283 f = h - i; // factorial part of h
284 p = v * (1.0f - s);
285 q = v * (1.0f - s * f);
286 t = v * (1.0f - s * (1.0f - f));
287 switch (i) {
288 case 0:
289 r = v;
290 g = t;
291 b = p;
292 break;
293 case 1:
294 r = q;
295 g = v;
296 b = p;
297 break;
298 case 2:
299 r = p;
300 g = v;
301 b = t;
302 break;
303 case 3:
304 r = p;
305 g = q;
306 b = v;
307 break;
308 case 4:
309 r = t;
310 g = p;
311 b = v;
312 break;
313 default: // case 5:
314 r = v;
315 g = p;
316 b = q;
317 break;
318 }
319
320 return ((uint8_t)(r * 255.0f) << 16) | ((uint8_t)(g * 255.0f) << 8) | (uint8_t)(b * 255.0f);
321 }

331 inline void toHSV(color8888 c, float_t& h, float_t& s, float_t& v) {
332 float_t cmin, cmax, delta;
333 float_t r = red(c) / 255.0f;
334 float_t g = green(c) / 255.0f;
335 float_t b = blue(c) / 255.0f;
336 cmin = min(min(r, g), b);
337 cmax = max(max(r, g), b);
338 v = cmax;
339 delta = cmax - cmin;
340 if (cmax != 0) s = delta / cmax;
341 else {
342 // r = g = b = 0 // s = 0, v is undefined
343 v = 0;
344 s = 0;
345 h = 0;
346 return;
347 }
348 if (r == cmax)
349 h = delta == 0 ? 0 : (g - b) / delta; // between yellow & magenta
350 else if (g == cmax)
351 h = delta == 0 ? 0 : 2 + (b - r) / delta; // between cyan & yellow
352 else
353 h = delta == 0 ? 0 : 4 + (r - g) / delta; // between magenta & cyan
354 h *= 60; // degrees
355 if (h < 0) h += 360;
356 }

365 inline color888 blend888(color888 fg, color888 bg, alpha_t alpha) {
366 uint8_t a = alpha8bit(alpha);
367 uint32_t rb = fg & 0xff00ff;
368 uint32_t g = fg & 0x00ff00;
369 rb += ((bg & 0xff00ff) - rb) * a >> 8;
370 g += ((bg & 0x00ff00) - g) * a >> 8;
371 return (rb & 0xff00ff) | (g & 0xff00);
372 }
