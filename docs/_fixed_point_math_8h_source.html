342 Number operator * (Number& other) {
343 Number a(*this);
344 Number b(other);
345 Number n(a.radixBits + b.radixBits);
346 unsigned char maxRadixBits = a.radixBits > b.radixBits ? a.radixBits : b.radixBits;
347
348 char neg = 0;
349 if (a.value < 0) {
350 neg = 1;
351 a.value = -a.value;
352 }
353 if (b.value < 0) {
354 neg = !neg;
355 b.value = -b.value;
356 }
357
358 FPM_TYPE h = ((a.value >> FPM_TYPE_BITS_2) * (b.value >> FPM_TYPE_BITS_2));
359 FPM_TYPE m1 = ((a.value >> FPM_TYPE_BITS_2) * (b.value & FPM_LOW_MASK));
360 FPM_TYPE m2 = ((b.value >> FPM_TYPE_BITS_2) * (a.value & FPM_LOW_MASK));
361 FPM_TYPE l = (((a.value & FPM_LOW_MASK) * (b.value & FPM_LOW_MASK)));
362
363 n.value = ((h << FPM_TYPE_BITS_2) + m1 + m2 + l) >> (n.radixBits - maxRadixBits);
364 n.radixBits = maxRadixBits;
365
366 //std::cout << "[a] val:" << a.value << " float:" << a.toFloat() << " rbits:" << int(a.radixBits) << std::endl;
